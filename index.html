    const form = document.getElementById('smellForm');

    form.addEventListener('submit', function(event) {

      const date = new Date().toLocaleDateString();

      const data = {

        date: date,

        rose: form.rose.value,

        girofle: form.girofle.value

      localStorage.setItem("odorat_progress", JSON.stringify(progression));


      result.innerHTML = `<p>✅ Séance enregistrée pour le ${date} !</p>`;

      form.reset();
